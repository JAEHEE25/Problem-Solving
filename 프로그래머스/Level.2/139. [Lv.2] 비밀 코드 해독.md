### [ 문제 풀이 ]
```Java
import java.util.*;
class Solution {
    static int[] code;
    static ArrayList<int[]> codes;
    static boolean[] visit;
    public int solution(int n, int[][] q, int[] ans) {
        //1~n 사이에 가능한 5개 조합 모두 구하기
        code = new int[5];
        codes = new ArrayList<>();
        visit = new boolean[n + 1];
        getCodes(1, 0, n);

        //q, ans 돌면서 안 맞는 조합 제거
        int answer = codes.size();
        for (int i = 0; i < codes.size(); i++) {
            for (int j = 0; j < q.length; j++) {
                int cnt = 0;
                for (int x = 0; x < 5; x++) {
                    for (int y = 0; y < 5; y++) {
                        if (codes.get(i)[x] == q[j][y]) {
                            cnt++;
                        }
                    }
                }

                if (cnt != ans[j]) { //개수가 다르면 해당 코드는 실패
                    answer--;
                    break;
                }
            }

        }
        
        return answer;
    }
    
    private void getCodes(int start, int depth, int n) {
        if (depth == 5) {
            codes.add(Arrays.copyOf(code, 5));
            return;
        }
        
        for (int i = start; i <= n; i++) {
            if (!visit[i]) {
                visit[i] = true;
                code[depth] = i;
                getCodes(i, depth + 1, n);
                visit[i] = false;
            }
        }
    }
}
```
