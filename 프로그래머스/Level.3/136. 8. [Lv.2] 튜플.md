```Java
import java.util.*;
class Solution {
    public int[] solution(String s) {
        //모든 숫자를 포함하는 Map
        Map<Integer, Integer> map = new HashMap<>();
        
        //우선순위 큐에 NumSet으로 만들어서 넣기
        PriorityQueue<NumSet> pq = new PriorityQueue<>();
        List<Integer> list = new ArrayList<>();
        StringBuilder sb = new StringBuilder();
        for (int i = 1; i < s.length() - 1; i++) {
            char c = s.charAt(i);
            if (c == '{') {
                list = new ArrayList<>();
            } else if (Character.isDigit(c)) {
                sb.append(c);
            } else if (c == ',' && s.charAt(i - 1) != '}') {
                int num = Integer.parseInt(sb.toString());
                list.add(num);
                map.put(num, 0);
                sb = new StringBuilder();
            } else if (c == '}') {
                int num = Integer.parseInt(sb.toString());
                list.add(num);
                map.put(num, 0);
                sb = new StringBuilder();
                NumSet set = new NumSet(list, list.size());
                pq.offer(set);
            }
        }
        
        //튜플 구하기
        int[] answer = new int[pq.size()];
        int idx = 0;
        while (!pq.isEmpty()) {
            List<Integer> pollList = pq.poll().list;
            for (int num : pollList) {
                if (map.get(num) == 0) {
                    answer[idx] = num;
                    map.put(num, 1);
                    idx++;
                }
            }
        }
        return answer;
    }
}

class NumSet implements Comparable<NumSet> {
    List<Integer> list;
    int size;
    
    NumSet(List<Integer> list, int size) {
        this.list = list;
        this.size = size;
    }
    
    @Override
    public int compareTo(NumSet o) { //size 기준으로 오름차순 정렬
        return this.size - o.size;
    }
    
}

```
```Java
import java.util.*;
class Solution {
    public int[] solution(String s) {
        Set<String> set = new HashSet<>();
        String[] arr = s.replaceAll("[{]", " ").replaceAll("[}]", " ").trim().split(" , ");
        Arrays.sort(arr, (a, b)->{return a.length() - b.length();});
        int[] answer = new int[arr.length];
        int idx = 0;
        for(String s1 : arr) {
            for(String s2 : s1.split(",")) {
                if(set.add(s2)) answer[idx++] = Integer.parseInt(s2);
            }
        }
        return answer;
    }
}
```
